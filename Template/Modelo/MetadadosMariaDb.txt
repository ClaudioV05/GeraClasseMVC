CREATE TABLE `usuarios` (
	`CODIGO` INT(11) NOT NULL AUTO_INCREMENT,
	`NOME` VARCHAR(80) NOT NULL COLLATE 'utf8mb4_general_ci',
	`SOBRENOME` VARCHAR(100) NOT NULL COLLATE 'utf8mb4_general_ci',
	`IDADE` INT(11) NOT NULL,
	`SEXO` CHAR(1) NULL DEFAULT 'M' COLLATE 'utf8mb4_general_ci',
	`ATIVO` CHAR(1) NULL DEFAULT 'T' COLLATE 'utf8mb4_general_ci',
	PRIMARY KEY (`CODIGO`) USING BTREE,
	UNIQUE INDEX `IDX_COD_USUARIOS` (`CODIGO`) USING BTREE,
	CONSTRAINT `CODDADOSCEP` FOREIGN KEY (`CODIGO`) REFERENCES `crud_viacep`.`dadoscep` (`CODIGO`) ON UPDATE NO ACTION ON DELETE NO ACTION
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=2
;


CREATE TABLE `dadoscep` (
	`CODIGO` INT(11) NOT NULL AUTO_INCREMENT,
	`CEP` INT(11) NULL DEFAULT NULL,
	`LOGRADOURO` VARCHAR(80) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`COMPLEMENTO` VARCHAR(80) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`BAIRRO` VARCHAR(100) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`LOCALIDADE` VARCHAR(80) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`UF` CHAR(2) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	`IBGE` VARCHAR(50) NULL DEFAULT NULL COLLATE 'utf8mb4_general_ci',
	PRIMARY KEY (`CODIGO`) USING BTREE,
	UNIQUE INDEX `IDX_COD_DADOSCEP` (`CODIGO`) USING BTREE
)
COLLATE='utf8mb4_general_ci'
ENGINE=InnoDB
AUTO_INCREMENT=2
;
